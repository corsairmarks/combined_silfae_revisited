# fired when the game starts
on_game_start = {
	events = {
		octeelan_evt.13 # flag mod as installed
	}
}

# fired when the game starts for each country
# THIS = country
on_game_start_country = {
	events = {
		octeelan_evt.3
		octeelan_evt.5
	}
}

# fired when a new leader is generated for an empire, to be available for recruitment
# THIS = country
# FROM = leader

	events = {
		octeelan_evt.1
	}
}

# fired when an election starts
# THIS = country
on_election_started = {
	events = {
		octeelan_evt.1
	}
}

# fired during empire initialization when starting a new game
# THIS = planet, capital
# FROM = species, founder

	events = {

	}
