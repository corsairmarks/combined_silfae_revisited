ruler_job_check_trigger = {
	if = {
		limit = {
			exists = owner
			owner = {
				has_origin = origin_necrophage
				has_trait = trait_necrophage
			}
		}
		custom_tooltip = {
			text = RULER_JOB_NECROPHAGE_TRIGGER
			has_trait = trait_necrophage
			is_enslaved = no
			is_being_purged = no
			is_being_assimilated = no
		}
	}
	else_if = {
		limit = {
			exists = owner
			owner = {
				has_origin = origin_holofrixit
				has_trait = trait_holooverseer
			}
		}
		custom_tooltip = {
			text = RULER_JOB_HOLOOVERSEER_TRIGGER
			hidden_trigger = {
				is_enslaved = no
				is_being_purged = no
				is_being_assimilated = no
				NOR = {
					has_trait = trait_syncretic_proles
					has_trait = trait_nerve_stapled
					has_citizenship_type = { type = citizenship_organic_trophy }
				}
				OR = {
					has_trait = trait_holooverseer
					AND = {
						owner = {
							has_technology = tech_synthetic_workers
							has_policy_flag = ai_full_rights
						}
						OR = {
							has_trait = trait_holofrixigram
							has_trait = trait_holofrixigram_2
							has_trait = trait_holofrixigram_3
						}
					}
				}
				if = {
					limit = { divinity_right_to_work_job_check_trigger_exempt = no }
					divinity_right_to_work_job_check_trigger_ruler = yes
				}
			}
		}
	}
	else = {
		custom_tooltip = RULER_JOB_TRIGGER
		hidden_trigger = {
			exists = owner
			is_enslaved = no
			is_being_purged = no
			is_being_assimilated = no
			NOR = {
				has_trait = trait_syncretic_proles
				has_trait = trait_nerve_stapled
				has_citizenship_type = { type = citizenship_organic_trophy }
			}
			OR = {
				NOT = { has_trait = trait_mechanical }
				owner = {
					has_technology = tech_synthetic_workers
					has_policy_flag = ai_full_rights
				}
			}
			if = {
				limit = { divinity_right_to_work_job_check_trigger_exempt = no }
				divinity_right_to_work_job_check_trigger_ruler = yes
			}
		}
	}
}

complex_specialist_job_check_trigger = {
	specialist_job_check_trigger = yes
	hidden_trigger = {
		exists = owner
		NOR = {
			has_trait = trait_holowarrior
			has_trait = trait_holowarrior_2
			has_trait = trait_holodrone
		}
		OR = {
			NOT = { has_trait = trait_mechanical }
			owner = {
				has_technology = tech_droid_workers
				NOT = { has_policy_flag = ai_outlawed }
			}
		}
	}
}

is_organic_species = {
	species = {
		OR = {
			is_archetype = BIOLOGICAL
			is_archetype = LITHOID
			AND = {
				is_archetype = HOLOARCH
				NOR = {
					has_trait = trait_holofrixigram
					has_trait = trait_holofrixigram_2
					has_trait = trait_holofrixigram_3
				}
			}
		}
	}
}

is_robotic_species = {
	species = {
		OR = {
			is_robotic = yes
			AND = {
				is_archetype = HOLOARCH
				OR = {
					has_trait = trait_holofrixigram
					has_trait = trait_holofrixigram_2
					has_trait = trait_holofrixigram_3
				}
			}
		}
	}
}
