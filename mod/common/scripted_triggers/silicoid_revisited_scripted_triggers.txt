# scope = anything with species (country/species/pop)
has_silicoid_food_upkeep = {
	optimize_memory
	species = {
		has_trait = trait_silicoid
		NOT = { has_trait = trait_voidling }
	}
}

# scope = anything with species (country/species/pop)
has_silicoid_mineral_upkeep = {
	optimize_memory
	species = {
		has_trait = trait_silicoid
		NOR = {
			has_trait = trait_voidling
			has_trait = trait_electroid
		}
	}
}

# this = species
has_electroid_energy_upkeep = {
	optimize_memory
	has_trait = trait_electroid
	NOR = {
		has_trait = trait_voidling
		planet = { is_planet_class = pc_nuked }
	}
}

			is_planet_class = pc_ocean

		}
	}
}
